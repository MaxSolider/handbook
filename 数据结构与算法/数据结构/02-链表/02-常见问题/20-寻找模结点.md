 static NormalListNode getLastModularNode(NormalListNode headNode, int k) {  
    if (headNode == null) {  
        return headNode;  
    }  
    NormalListNode modularNode = null;  
    int i = 1;  
    while (headNode != null) {  
        if (i % k == 0 ) {  
            modularNode = headNode;  
        }  
        headNode = headNode.getNext();  
        i++;  
    }  
    return modularNode;  
}
